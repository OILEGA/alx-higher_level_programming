import numpy as np

def matrix_mul(a, b):
    """
    Perform matrix multiplication between matrices a and b.
    """
    try:
        return np.dot(a, b)
    except ValueError as e:
        print("Error: {0}".format(e))
        return None


# Test 1: Correct usage
a = np.array([[1, 2], [3, 4]])
b = np.array([[5, 6], [7, 8]])
c = matrix_mul(a, b)
print("Test 1 - Result: \n{0}\n".format(c))

# Test 2: Error handling for non-square matrices
a = np.array([[1, 2, 3], [4, 5, 6]])
b = np.array([[7, 8], [9, 10], [11, 12]])
c = matrix_mul(a, b)
print("Test 2 - Result: {0}\n".format(c))

# Test 3: Error handling for matrices with incompatible dimensions
a = np.array([[1, 2], [3, 4], [5, 6]])
b = np.array([[7, 8], [9, 10]])
c = matrix_mul(a, b)
print("Test 3 - Result: {0}\n".format(c))
